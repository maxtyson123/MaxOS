121 template<class Type> Rectangle<Type> Rectangle<Type>::intersection(const Rectangle<Type>& other){
122 // If the rectangles don't intersect, return an empty rectangle
123 if(!intersects(other))
124 return Rectangle<Type>();
125
126 // Get the left and top of the intersection using the maximum of the two
127 Type left = this -> left > other.left ? this -> left : other.left;
128 Type top = this -> top > other.top ? this -> top : other.top;
129
130 // Get the right and bottom of the intersection using the minimum of the two
131 Type right = this -> left + this -> width < other.left + other.width ? this -> left + this -> width : other.left + other.width;
132 Type bottom = this -> top + this -> height < other.top + other.height ? this -> top + this -> height : other.top + other.height;
133
134 // Return the intersection
135 return Rectangle<Type>(left, top, right - left, bottom - top);
