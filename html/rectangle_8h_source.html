108 template<class Type> Rectangle<Type> Rectangle<Type>::intersection(const Rectangle<Type>& other){
109 // If the rectangles don't intersect, return an empty rectangle
110 if(!intersects(other))
111 return Rectangle<Type>();
112
113 // Get the left and top of the intersection using the maximum of the two
114 Type left = this -> left > other.left ? this -> left : other.left;
115 Type top = this -> top > other.top ? this -> top : other.top;
116
117 // Get the right and bottom of the intersection using the minimum of the two
118 Type right = this -> left + this -> width < other.left + other.width ? this -> left + this -> width : other.left + other.width;
119 Type bottom = this -> top + this -> height < other.top + other.height ? this -> top + this -> height : other.top + other.height;
120
121 // Return the intersection
122 return Rectangle<Type>(left, top, right - left, bottom - top);
123 }

132 template<class Type> Vector<Rectangle<Type> > Rectangle<Type>::subtract(const Rectangle<Type>& other){
133
134 // Store the result rectangle
135 Vector<Rectangle<Type> > result;
136
137 // Make sure the rectangles intersect
138 if(!intersects(other))
139 {
140 result.push_back(*this);
141 return result;
142 }
143
144 // Get the minimum and maximum values for the width and height
145 Type maxLeft = left > other.left ? left : other.left;
146 Type minRight = (left + width < other.left + other.width ? left + width : other.left + other.width);
147 Type minBottom = (top + height < other.top + other.height ? top+height : other.top + other.height);
148
149 // Add the non-intersecting rectangles to the result
150
151 // Add non-overlapping region above if current top is less than other top
152 if (top < other.top)
153 result.push_back(Rectangle<Type>(maxLeft, top, minRight - maxLeft,
154 other.top - top));
155
156 // Add non-overlapping region to the left if current left is less than other left
157 if (left < other.left)
158 result.push_back(
159 Rectangle<Type>(left, top, other.left - left, height));
160
161 // Add non-overlapping region to the right if current right is greater than other right
162 if (left + width > other.left + other.width)
163 result.push_back(Rectangle<Type>(
164 other.left + other.width, top,
165 (left + width) - (other.left + other.width) + 1, height));
166
167 // Add non-overlapping region below if current bottom is greater than other bottom
168 if (this->top + this->height > other.top + other.height)
169 result.push_back(Rectangle<Type>(maxLeft, minBottom,
170 minRight - maxLeft,
171 top + height - minBottom + 1));
