83MouseEventHandler* Window::on_mouse_button_pressed(uint32_t mouseX, uint32_t mouseY, uint8_t button) {
84
85 // Pass the mouse event to the children
86 drivers::peripherals::MouseEventHandler* child_result = CompositeWidget::on_mouse_button_pressed(mouseX, mouseY, button);
87 Rectangle<int32_t> window_position = position();
88
89 // Bring the window to the front
90 bring_to_front();
91
92 // Convert the mouse coordinates to an int32_t
93 auto x = (int32_t) mouseX;
94 auto y = (int32_t) mouseY;
95
96 if (x <= frame_thickness) {
97 if (y <= frame_thickness)
98 return &m_resizer_top_left;
99
100 else if (y < window_position.height - frame_thickness)
101 return &m_resizer_left;
102
103 else
104 return &m_resizer_bottom_left;
105 } else if (x < window_position.width - frame_thickness) {
106 if (y <= frame_thickness)
107 return &m_resizer_top;
108
109 else if (y < frame_thickness + title_bar_height)
110 return &m_mover;
111
112 else if (y >= window_position.height - frame_thickness)
113 return &m_resizer_bottom;
114 } else {
115 if (y <= frame_thickness)
116 return &m_resizer_top_right;
117
118 else if (y < window_position.height - frame_thickness)
119 return &m_resizer_right;
120
121 else
122 return &m_resizer_bottom_right;
123 }
124
125 return child_result;
126}

133void Window::draw_self(common::GraphicsContext* gc, common::Rectangle<int32_t>& area) {
134
135 // Get the positioning of the window
136 Coordinates window_absolute_position = CompositeWidget::absolute_coordinates(Coordinates(0, 0));
137 Rectangle<int32_t> windowPosition = this->position();
138 int32_t window_x = window_absolute_position.first;
139 int32_t window_y = window_absolute_position.second;
140
141 // Create an area for the window contents
142 Rectangle<int32_t> window_contents_area(frame_thickness, frame_thickness + title_bar_height,
143 windowPosition.width - 2 * frame_thickness,
144 windowPosition.height - 2 * frame_thickness - title_bar_height);
145
146 // Draw the window contents if they are in the area to draw
147 if (window_contents_area.intersects(area)) {
148 Rectangle<int32_t> contents_drawable = window_contents_area.intersection(area);
149 gc->fill_rectangle(contents_drawable.left + window_x, contents_drawable.top + window_y,
150 contents_drawable.left + contents_drawable.width + window_x,
151 contents_drawable.top + contents_drawable.height + window_y, area_colour);
152 }
153
154 // Draw the frame if it is in the area to draw
155 Rectangle<int32_t> window_frame_top_area(frame_thickness, 0, windowPosition.width - 2 * frame_thickness,
156 frame_thickness + title_bar_height);
157 if (window_frame_top_area.intersects(area)) {
158 Rectangle<int32_t> frame_drawable = window_frame_top_area.intersection(area);
159 gc->fill_rectangle(frame_drawable.left + window_x, frame_drawable.top + window_y,
160 frame_drawable.left + frame_drawable.width + window_x,
161 frame_drawable.top + frame_drawable.height + window_y, frame_colour);
162 }
163
164 // Draw the bottom of the window frame
165 Rectangle<int32_t> window_frame_bottom_area(frame_thickness, windowPosition.height - frame_thickness,
166 windowPosition.width - 2 * frame_thickness, frame_thickness);
167 if (window_frame_bottom_area.intersects(area)) {
168 Rectangle<int32_t> bottom_drawable = window_frame_bottom_area.intersection(area);
169 gc->fill_rectangle(window_x + bottom_drawable.left, window_y + bottom_drawable.top,
170 window_x + bottom_drawable.left + bottom_drawable.width,
171 window_y + bottom_drawable.top + bottom_drawable.height, frame_colour);
172 }
173
174 // Draw the left of the window frame
175 Rectangle<int32_t> window_frame_left_area(0, 0, frame_thickness, windowPosition.height);
176 if (window_frame_left_area.intersects(area)) {
177 Rectangle<int32_t> left_drawable = window_frame_left_area.intersection(area);
178 gc->fill_rectangle(window_x + left_drawable.left, window_y + left_drawable.top,
179 window_x + left_drawable.left + left_drawable.width,
180 window_y + left_drawable.top + left_drawable.height, frame_colour);
181 }
182
183 // Draw the right of the window frame
184 Rectangle<int32_t> window_frame_right_area(windowPosition.width - frame_thickness, 0, frame_thickness,
185 windowPosition.height);
186 if (window_frame_right_area.intersects(area)) {
187 Rectangle<int32_t> right_drawable = window_frame_right_area.intersection(area);
188 gc->fill_rectangle(window_x + right_drawable.left, window_y + right_drawable.top,
189 window_x + right_drawable.left + right_drawable.width,
190 window_y + right_drawable.top + right_drawable.height, frame_colour);
191 }
192}
