105 // Pass the mouse event to the children
106 drivers::peripherals::MouseEventHandler* child_result = CompositeWidget::on_mouse_button_pressed(mouse_x, mouse_y, button);

108
109 // Bring the window to the front
110 bring_to_front();
111
112 // Convert the mouse coordinates to an int32_t

116 if (x <= frame_thickness) {

119
120 else if (y < window_position.height - frame_thickness)
121 return &m_resizer_left;
122
123 else
124 return &m_resizer_bottom_left;
125 } else if (x < window_position.width - frame_thickness) {
126 if (y <= frame_thickness)
127 return &m_resizer_top;
128
129 else if (y < frame_thickness + title_bar_height)
130 return &m_mover;
131
132 else if (y >= window_position.height - frame_thickness)
133 return &m_resizer_bottom;
134 } else {
135 if (y <= frame_thickness)
136 return &m_resizer_top_right;
137
138 else if (y < window_position.height - frame_thickness)
139 return &m_resizer_right;
140
141 else

154void Window::draw_self(common::GraphicsContext* gc, common::Rectangle<int32_t>& area) {
155
156 // Get the positioning of the window
157 Coordinates window_absolute_position = CompositeWidget::absolute_coordinates(Coordinates(0, 0));
158 Rectangle<int32_t> window_position = this->position();
159 int32_t window_x = window_absolute_position.first;
160 int32_t window_y = window_absolute_position.second;
161
162 // Create an area for the window contents
163 Rectangle<int32_t> window_contents_area(frame_thickness, frame_thickness + title_bar_height,
164 window_position.width - 2 * frame_thickness,
165 window_position.height - 2 * frame_thickness - title_bar_height);
166
167 // Draw the window contents if they are in the area to draw
168 if (window_contents_area.intersects(area)) {
169 Rectangle<int32_t> contents_drawable = window_contents_area.intersection(area);
170 gc->fill_rectangle(contents_drawable.left + window_x, contents_drawable.top + window_y,
171 contents_drawable.left + contents_drawable.width + window_x,
172 contents_drawable.top + contents_drawable.height + window_y, area_colour);
173 }
174
175 // Draw the frame if it is in the area to draw
176 Rectangle<int32_t> window_frame_top_area(frame_thickness, 0, window_position.width - 2 * frame_thickness,
177 frame_thickness + title_bar_height);
178 if (window_frame_top_area.intersects(area)) {
179 Rectangle<int32_t> frame_drawable = window_frame_top_area.intersection(area);
180 gc->fill_rectangle(frame_drawable.left + window_x, frame_drawable.top + window_y,
181 frame_drawable.left + frame_drawable.width + window_x,
182 frame_drawable.top + frame_drawable.height + window_y, frame_colour);
183 }
184
185 // Draw the bottom of the window frame
186 Rectangle<int32_t> window_frame_bottom_area(frame_thickness, window_position.height - frame_thickness,
187 window_position.width - 2 * frame_thickness, frame_thickness);
188 if (window_frame_bottom_area.intersects(area)) {
189 Rectangle<int32_t> bottom_drawable = window_frame_bottom_area.intersection(area);
190 gc->fill_rectangle(window_x + bottom_drawable.left, window_y + bottom_drawable.top,
191 window_x + bottom_drawable.left + bottom_drawable.width,
192 window_y + bottom_drawable.top + bottom_drawable.height, frame_colour);
193 }
194
195 // Draw the left of the window frame

197 if (window_frame_left_area.intersects(area)) {
198 Rectangle<int32_t> left_drawable = window_frame_left_area.intersection(area);
