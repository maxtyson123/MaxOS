57Event<EthernetDriverEvents>* EthernetDriverEventHandler::on_event(Event<EthernetDriverEvents>* event) {
58
59 switch (event->type) {
60
61 case EthernetDriverEvents::BEFORE_SEND:
62 before_send(((BeforeSendEvent*) event)->buffer, ((BeforeSendEvent*) event)->size);
63 break;
64
65 case EthernetDriverEvents::DATA_SENT:
66 data_sent(((DataSentEvent*) event)->buffer, ((DataSentEvent*) event)->size);
67 break;
68
69 case EthernetDriverEvents::DATA_RECEIVED:
70 event->return_value.bool_value = data_received(((DataReceivedEvent*) event)->buffer,
71 ((DataReceivedEvent*) event)->size);
72 break;
73
74 default:
75 break;
76 }
77

79}

122void EthernetDriver::fire_data_received(uint8_t* buffer, uint32_t size) {
123
124 // Raise the event
125 Vector<Event<EthernetDriverEvents>*> values =
126 raise_event(new DataReceivedEvent(buffer, size));
127
128 // Loop through the events
129 for (auto &value : values) {
130 switch (value->type) {
131 case EthernetDriverEvents::DATA_RECEIVED:
132 if (value->return_value.bool_value)
133 send(buffer, size);
134 break;
135
136 default:
137 break;
138 }
139 }
140}
