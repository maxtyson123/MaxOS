48Event<EthernetDriverEvents>* EthernetDriverEventHandler::on_event(Event<EthernetDriverEvents>* event) {
49
50 switch (event->type) {
51
52 case EthernetDriverEvents::BEFORE_SEND:

55
56 case EthernetDriverEvents::DATA_SENT:
57 DataSent(((DataSentEvent*) event)->buffer, ((DataSentEvent*) event)->size);

66 break;
67 }

110void EthernetDriver::FireDataReceived(uint8_t* buffer, uint32_t size) {
111
112 // Raise the event
113 Vector<Event<EthernetDriverEvents>*> values =
114 raise_event(new DataReceivedEvent(buffer, size));
115
116 // Loop through the events
117 for (auto &value : values) {
118 switch (value->type) {
119 case EthernetDriverEvents::DATA_RECEIVED:
120 if (value->return_value.boolValue)
121 Send(buffer, size);
122 break;
123
124 default:
125 break;
126 }
