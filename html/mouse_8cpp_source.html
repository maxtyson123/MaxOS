120 void MouseDriver::handle_interrupt(){
121 
122  //Only if the 6th bit of data is one then there is data to handle
123  uint8_t status = command_port.read();
124  if(!(status & 0x20))
125  return;
126 
127  // read the data and store it in the buffer
128  buffer[offset] = data_port.read();
129  offset = (offset + 1) % 3;
130 
131  // If the mouse data transmission is incomplete (3rd piece of data isn't through)
132  if(offset != 0)
133  return;
134 
135  // If the mouse is moved (y-axis is inverted)
136  if(buffer[1] != 0 || buffer[2] != 0)
137  raise_event(new MouseMoveEvent(buffer[1], -buffer[2]));

140

143  {
144  // Check if the button is up or down
