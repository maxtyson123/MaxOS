85void MouseDriver::activate() {
86
87 // Get the current state of the mouse
88 command_port.write(0x20);
89 uint8_t status = (data_port.read() | 2);
90
91 // Write the new state
92 command_port.write(0x60);
93 data_port.write(status);
94
95 // Tell the PIC to start listening to the mouse
96 command_port.write(0xAB);
97
98 // Activate the mouse
99 command_port.write(0xD4);
100 data_port.write(0xF4);
101 data_port.read();
102}

107void MouseDriver::handle_interrupt() {
108
109 // Check if there is data to handle
110 uint8_t status = command_port.read();
111 if (!(status & 0x20))
112 return;
113
114 // Read the data
115 m_buffer[m_offset] = data_port.read();
116 m_offset = (m_offset + 1) % 3;
117
118 // If the mouse data transmission is incomplete (3rd piece of data isn't through)
119 if (m_offset != 0)
120 return;
121
122 // If the mouse is moved (y-axis is inverted)
123 if (m_buffer[1] != 0 || m_buffer[2] != 0)
124 raise_event(new MouseMoveEvent(m_buffer[1], -m_buffer[2]));
125
126 // Handle button presses
127 for (int i = 0; i < 3; ++i) {
128
129 // This button is still in the same state
130 if ((m_buffer[0] & (0x1 << i)) == (m_buttons & (0x1 << i)))
131 continue;
132
133 // Pass to handlers
134 bool is_pressed = (m_buttons & (0x1 << i)) != 0;
135 if (is_pressed)
136 raise_event(new MouseDownEvent(m_buffer[i]));
137 else
138 raise_event(new MouseUpEvent(m_buffer[i]));
139 }
140
141 m_buttons = m_buffer[0];
142}
