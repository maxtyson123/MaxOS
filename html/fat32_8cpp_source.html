16Fat32Volume::Fat32Volume(Disk *hd, uint32_t partition_offset)
17: disk(hd)
18{
19
20 // Read the BIOS parameter block
21 buffer_t bpb_buffer(&bpb, sizeof(bpb32_t));
22 disk->read(partition_offset, &bpb_buffer);
23
24 // Parse the FAT info
25 uint32_t total_data_sectors = bpb.total_sectors_32 - (bpb.reserved_sectors + (bpb.table_copies * bpb.table_size_32));
26 fat_total_clusters = total_data_sectors / bpb.sectors_per_cluster;
27 fat_lba = partition_offset + bpb.reserved_sectors;
28 fat_copies = bpb.table_copies;
29 fat_info_lba = partition_offset + bpb.fat_info;
30 data_lba = fat_lba + (bpb.table_copies * bpb.table_size_32);
31 root_lba = data_lba + bpb.sectors_per_cluster * (bpb.root_cluster - 2);
32
33 // Read the fs info
34 buffer_t fs_buffer(&fsinfo, sizeof(fs_info_t));
35 disk->read(fat_info_lba, &fs_buffer);
36
37 // Validate the fat information
38 if (fsinfo.lead_signature != 0x41615252 || fsinfo.structure_signature != 0x61417272 ||
39 fsinfo.trail_signature != 0xAA550000) {
40 Logger::ERROR() << "Invalid FAT32 filesystem information TODO: Handle this\n";
41 return;
42 }
43}

54
55 // Get the location in the FAT table
56 lba_t offset = cluster * sizeof(uint32_t);
57 lba_t sector = fat_lba + (offset / bpb.bytes_per_sector);
58 uint32_t entry_index = offset % bpb.bytes_per_sector;
59
60 // Read the FAT entry

62 disk->read(sector, &fat);
63
64 // Get the next cluster info (mask the upper 4 bits)
65 auto entry = (uint32_t *) (&(fat.raw()[entry_index])); // TODO & here is weird
66 return *entry & 0x0FFFFFFF;

76uint32_t Fat32Volume::set_next_cluster(uint32_t cluster, uint32_t next_cluster) {
77
78 // TODO - when in userspace: For performance cache fat entirely, cache file data, cache cluster chains
79
80 // Get the location in the FAT table
81 lba_t offset = cluster * sizeof(uint32_t);
82
83 for (int i = 0; i < fat_copies; ++i) {
84
85 lba_t sector = (fat_lba + i * bpb.table_size_32) + (offset / bpb.bytes_per_sector);
86 uint32_t entry_index = offset % bpb.bytes_per_sector;
87
88 // Read the FAT entry
89 buffer_t fat(bpb.bytes_per_sector);
90 disk->read(sector, &fat);
91
92 // Set the next cluster info (mask the upper 4 bits)
93 auto entry = (uint32_t *) (&(fat.raw()[entry_index]));
94 *entry = next_cluster & 0x0FFFFFFF;
95 disk->write(sector, &fat);
96
97 }
98
99 return next_cluster;

107uint32_t Fat32Volume::find_free_cluster() {
108
109 // Get the first free cluster
110 for (uint32_t start = fsinfo.next_free_cluster; start < fat_total_clusters + 1; start++)
111 if (next_cluster(start) == 0)
112 return start;
113
114 // Check any clusters before the first free cluster
115 for (uint32_t start = 2; start < fsinfo.next_free_cluster; start++)
116 if (next_cluster(start) == 0)
117 return start;
118
119 ASSERT(false, "No free clusters found in the FAT table");
120 return 0;
121}

142uint32_t Fat32Volume::allocate_cluster(uint32_t cluster, size_t amount) {
143
144 // Make sure within bounds
145 if (cluster > fat_total_clusters || cluster + amount > fat_total_clusters)
146 return 0;
147
148 // Go through allocating the clusters
149 for (size_t i = 0; i < amount; i++) {
150 uint32_t next_cluster = find_free_cluster();
151
152 // Update the fsinfo
153 fsinfo.next_free_cluster = next_cluster + 1;
154 fsinfo.free_cluster_count -= 1;
155
156 // If there is an existing chain it needs to be updated
157 if (cluster != 0)
158 set_next_cluster(cluster, next_cluster);
159
160 cluster = next_cluster;
161 }
162
163 // Once all the updates are done flush the changes to the disk
164 buffer_t fs_info_buffer(&fsinfo, sizeof(fs_info_t));
165 disk->write(fat_info_lba, &fs_info_buffer);
166
167 // Finish the chin
168 set_next_cluster(cluster, (uint32_t) ClusterState::END_OF_CHAIN);
169 uint32_t next = next_cluster(cluster);
170 return cluster;
171}

192void Fat32Volume::free_cluster(uint32_t cluster, size_t amount) {
193
194 // Make sure within bounds
195 if (cluster < 2 || cluster > fat_total_clusters || cluster + amount > fat_total_clusters)
196 return;
197
198 // Go through freeing the clusters
199 for (size_t i = 0; i < amount; i++) {
200
201 // Find the next cluster before it is removed from the chain
202 uint32_t next_in_chain = next_cluster(cluster);
203
204 // Update the fsinfo
205 fsinfo.next_free_cluster = cluster;
206 fsinfo.free_cluster_count += 1;
207
208 // Update the chain
209 set_next_cluster(cluster, (lba_t) ClusterState::FREE);
210 cluster = next_in_chain;
211 }
212
213 // Save the fsinfo
214 buffer_t fs_info_buffer(&fsinfo, sizeof(fs_info_t));
215 disk->write(fat_info_lba, &fs_info_buffer);
216
217 // Mark the end of the chain
218 set_next_cluster(cluster, (uint32_t) ClusterState::END_OF_CHAIN);
219}

242void Fat32File::write(const buffer_t *data, size_t amount) {
243
244 size_t buffer_space = m_volume->bpb.bytes_per_sector * m_volume->bpb.sectors_per_cluster;
245 buffer_t buffer(buffer_space);
246 buffer.clear();
247
248 uint64_t current_offset = 0;
249 uint64_t bytes_written = 0;
250 uint32_t last = m_first_cluster;
251
252 // Read the file
253 for (uint32_t cluster = last;
254 cluster != (uint32_t) ClusterState::END_OF_CHAIN; cluster = m_volume->next_cluster(cluster)) {
255 last = cluster;
256
257 // No cluster to read from (blank file)
258 if (cluster == 0)
259 break;
260
261 // Skip clusters before the offset
262 if ((current_offset + buffer_space) < m_offset) {
263 current_offset += buffer_space;
264 continue;
265 }
266
267 // Read each sector in the cluster (prevent overwriting the data)
268 lba_t lba = m_volume->data_lba + (cluster - 2) * m_volume->bpb.sectors_per_cluster;
269 for (size_t sector = 0; sector < m_volume->bpb.sectors_per_cluster; sector++)
270 m_volume->disk->read(lba + sector, &buffer, m_volume->bpb.bytes_per_sector);
271 buffer.set_offset(0);
272
273 // If the offset is in the middle of the cluster
274 size_t buffer_offset = 0;
275 if (m_offset > current_offset)
276 buffer_offset = m_offset - current_offset;
277
278 // Calculate how many bytes are being copied (read from cluster at offset?
279 // or read part of cluster?)
280 size_t cluster_remaining_bytes = buffer_space - buffer_offset;
281 size_t data_remaining_bytes = amount - bytes_written;
282 size_t bytes_to_copy = (cluster_remaining_bytes < data_remaining_bytes) ? cluster_remaining_bytes
283 : data_remaining_bytes;
284 bytes_to_copy = (bytes_to_copy > buffer_space) ? buffer_space : bytes_to_copy;
285
286 // Update the data
287 buffer.copy_from(data, bytes_to_copy, buffer_offset, bytes_written);
288 bytes_written += bytes_to_copy;
289 current_offset += bytes_to_copy;
290 buffer.set_offset(0);
291
292 // Write the data back to the disk
293 for (size_t sector = 0; sector < m_volume->bpb.sectors_per_cluster; sector++)
294 m_volume->disk->write(lba + sector, &buffer, m_volume->bpb.bytes_per_sector);
295 }
296
297 // Extend the file
298 while (bytes_written < amount) {
299 // Allocate a new cluster
300 uint32_t new_cluster = m_volume->allocate_cluster(last);
301 if (new_cluster == 0)
302 break;
303
304 if (last == 0)
305 m_first_cluster = new_cluster;
306
307 // Update the data
308 size_t bytes_to_copy = (amount - bytes_written) > buffer_space ? buffer_space : (amount - bytes_written);
309 buffer.copy_from(data, bytes_to_copy, 0, bytes_written);
310 bytes_written += bytes_to_copy;
311 current_offset += bytes_to_copy;
312 buffer.set_offset(0);
313
314 // Write the data back to the disk
315 lba_t lba = m_volume->data_lba + (new_cluster - 2) * m_volume->bpb.sectors_per_cluster;
316 for (size_t sector = 0; sector < m_volume->bpb.sectors_per_cluster; sector++)
317 m_volume->disk->write(lba + sector, &buffer, m_volume->bpb.bytes_per_sector);
318
319 // Go to the next cluster
320 last = new_cluster;
321 }
322
323 // Update file size
324 m_offset += bytes_written;
325 if (m_offset > m_size)
326 m_size = m_offset;
327
328 // Update entry info
329 m_entry->size = m_size;
330 m_entry->first_cluster_high = (m_first_cluster >> 16) & 0xFFFF;
331 m_entry->first_cluster_low = m_first_cluster & 0xFFFF;
332 // TODO: When implemented as a usermode driver save the time
333 m_parent_directory->save_entry_to_disk(m_entry);
334
335}

415dir_entry_t *Fat32Directory::create_entry(const string &name, bool is_directory) {
416
417 // Allocate a cluster for the new entry
418 uint32_t cluster = m_volume->allocate_cluster(0);
419 if (cluster == 0)
420 return nullptr;
421
422 // Store the name
423 Vector<long_file_name_entry_t> lfn_entries = to_long_filenames(name);
424 char short_name[8];
425 char short_extension[3];
426 for (int i = 0; i < 8; i++)
427 short_name[i] = (i < name.length()) ? name[i] : ' ';
428 for (int i = 0; i < 3; i++)
429 short_extension[i] = (8 + i < name.length()) ? name[8 + i] : ' ';
430
431 // Create the directory entry
432 dir_entry_t entry = {};
433 memcpy(entry.name, short_name, sizeof(short_name));
434 memcpy(entry.extension, short_extension, sizeof(short_extension));
435 entry.attributes = is_directory ? (uint8_t) DirectoryEntryAttributes::DIRECTORY
436 : (uint8_t) DirectoryEntryAttributes::ARCHIVE;
437 entry.first_cluster_high = (cluster >> 16) & 0xFFFF;
438 entry.first_cluster_low = cluster & 0xFFFF;
439
440 // Find free space for the new entry and the name
441 size_t entries_needed = 1 + lfn_entries.size();
442 int entry_index = find_free_entries(entries_needed);
443 if (entry_index == -1)
444 entry_index = expand_directory(entries_needed);
445
446 // Store the entries in the cache
447 for (size_t i = 0; i < entries_needed; i++)
448 m_entries[entry_index + i] = i == 0 ? entry : *(dir_entry_t *) &lfn_entries[i - 1];
449
450 // Write the long file name entries
451 for (size_t index = entry_index + lfn_entries.size() - 1; index >= entry_index; index--)
452 update_entry_on_disk(index);
453
454 // Creating the file is done
455 if (!is_directory)
456 return &m_entries[entry_index];
457
458 // Create the "." in the directory
459 dir_entry_t current_dir_entry = {};
460 memcpy(current_dir_entry.name, ".", 1);
461 current_dir_entry.attributes = 0x10;
462 current_dir_entry.first_cluster_high = (cluster >> 16) & 0xFFFF;
463 current_dir_entry.first_cluster_low = cluster & 0xFFFF;
464
465 // Create the ".." in the directory
466 dir_entry_t parent_dir_entry = {};
467 memcpy(parent_dir_entry.name, "..", 2);
468 parent_dir_entry.attributes = 0x10;
469 parent_dir_entry.first_cluster_high = (m_first_cluster >> 16) & 0xFFFF;
470 parent_dir_entry.first_cluster_low = m_first_cluster & 0xFFFF;
471
472 // Write the entries to the disk
473 uint32_t bytes_per_sector = m_volume->bpb.bytes_per_sector;
474 lba_t child_lba = m_volume->data_lba + (cluster - 2) * bytes_per_sector;
475 buffer_t buffer(bytes_per_sector);
476 buffer.clear();
477 buffer.copy_from(&current_dir_entry, sizeof(dir_entry_t));
478 buffer.copy_from(&parent_dir_entry, sizeof(dir_entry_t));
479 m_volume->disk->write(child_lba, &buffer);
480
481 // Directory created
482 return &m_entries[entry_index];
483}

491void Fat32Directory::remove_entry(uint32_t cluster, const string &name) {
492
493 // Find the entry in the directory
494 int entry = entry_index(cluster);
495 if (entry == -1)
496 return;
497
498 // Find any long file name entries that belong to this entry
499 size_t delete_entry_index = entry;
500 while (delete_entry_index > 0 &&
501 m_entries[delete_entry_index - 1].attributes == (uint8_t) DirectoryEntryAttributes::LONG_NAME)
502 delete_entry_index--;
503
504 // Mark the entries as free
505 for (size_t i = delete_entry_index; i < entry; i++)
506 m_entries[i].name[0] = (uint8_t) DirectoryEntryType::FREE;
507
508 // Update the entries on the disk
509 for (size_t i = delete_entry_index; i <= entry; i++)
510 update_entry_on_disk(i);
511
512 // Count the number of clusters in the chain
513 size_t cluster_count = 0;
514 for (uint32_t next_cluster = cluster;
515 next_cluster < (lba_t) ClusterState::BAD; next_cluster = m_volume->next_cluster(next_cluster))
516 cluster_count++;
517
518 // Free all the clusters in the chain
519 m_volume->free_cluster(cluster, cluster_count);
520}

525void Fat32Directory::read_all_entries() {
526 m_entries.clear();
527
528 size_t buffer_space = m_volume->bpb.bytes_per_sector * m_volume->bpb.sectors_per_cluster;
529 buffer_t buffer(buffer_space);
530 buffer.clear();
531
532 // Read the directory
533 for (uint32_t cluster = m_first_cluster;
534 cluster != (uint32_t) ClusterState::END_OF_CHAIN; cluster = m_volume->next_cluster(cluster)) {
535
536 // Cache info to prevent re-traversing the chain
537 m_last_cluster = cluster;
538 m_current_cluster_length++;
539
540 // Read each sector in the cluster
541 lba_t lba = m_volume->data_lba + (cluster - 2) * m_volume->bpb.sectors_per_cluster;
542 for (size_t sector = 0; sector < m_volume->bpb.sectors_per_cluster; sector++)
543 m_volume->disk->read(lba + sector, &buffer, m_volume->bpb.bytes_per_sector);
544
545 // Parse the directory entries (each entry is 32 bytes)
546 for (size_t entry_offset = 0; entry_offset < buffer_space; entry_offset += 32) {
547
548 // Store the entry
549 auto entry = (dir_entry_t *) &(buffer.raw()[entry_offset]);
550 m_entries.push_back(*entry);
551
552 // Check if the entry is the end of the directory
553 if (entry->name[0] == (uint8_t) DirectoryEntryType::LAST)
554 return;
555 }
556
557 }
558}

578void Fat32Directory::update_entry_on_disk(int index) {
579
580 // Get the entry
581 auto entry = m_entries[index];
582
583 // Determine sector offset and in-sector byte offset
584 uint32_t bytes_per_sector = m_volume->bpb.bytes_per_sector;
585 uint32_t entry_offset = index * sizeof(dir_entry_t);
586 uint32_t sector_offset = entry_offset / bytes_per_sector;
587 uint32_t in_sector_offset = entry_offset % bytes_per_sector;
588
589 // Find which cluster has the entry
590 uint32_t cluster = m_first_cluster;
591 for (uint32_t offset_remaining = entry_offset;
592 offset_remaining >= bytes_per_sector; offset_remaining -= bytes_per_sector)
593 cluster = m_volume->next_cluster(cluster);
594
595 // Read the full sector into a buffer
596 lba_t base_lba = m_volume->data_lba + (cluster - 2) * m_volume->bpb.sectors_per_cluster;
597 buffer_t sector_buffer(bytes_per_sector, false);
598 m_volume->disk->read(base_lba + sector_offset, &sector_buffer);
599
600 // Update the entry in the buffer
601 sector_buffer.copy_from(&entry, sizeof(dir_entry_t), in_sector_offset);
602 m_volume->disk->write(base_lba + sector_offset, &sector_buffer);
603}

611int Fat32Directory::entry_index(lba_t cluster) {
612
613 int entry_index = 0;
614 for (; entry_index < m_entries.size(); entry_index++) {
615 auto &entry = m_entries[entry_index];
616
617 // End of directory means no more entries
618 if (entry.name[0] == (uint8_t) DirectoryEntryType::LAST)
619 return -1;
620
621 // Skip free entries
622 if (entry.name[0] == (uint8_t) DirectoryEntryType::FREE)
623 continue;
624
625 // Check if the entry is the one
626 uint32_t start_cluster = (entry.first_cluster_high << 16) | entry.first_cluster_low;
627 if (start_cluster == cluster)
628 break;
629 }
630
631 // Make sure the entry is valid
632 if (entry_index >= m_entries.size())
633 return -1;
634
635 return entry_index;
636
637}

646int Fat32Directory::find_free_entries(size_t amount) {
647
648 for (int entry_index = 0; entry_index < m_entries.size(); entry_index++) {
649 // Check if there are enough free entries in a row

652 if (m_entries[entry_index + j].name[0] != (char) DirectoryEntryType::FREE)
653 found = false;
654

657 }

659 return -1;

673int Fat32Directory::expand_directory(size_t amount) {
674
675 // Remove the old end of directory marker
676 int free_start = m_entries.size() - 1;
677 ASSERT(m_entries[free_start].name[0] == (uint8_t) DirectoryEntryType::LAST, "Last entry is not marked");
678 m_entries[free_start].name[0] = (uint8_t) DirectoryEntryType::FREE;
679
680 // Count how many free entries there is before the end
681 for (int i = free_start; i >= 0; --i) {
682 if (m_entries[i].name[0] == (char) DirectoryEntryType::FREE)
683 free_start = i;
684 else
685 break;
686 }
687
688 // Calculate how many entries are need to be created (ie was there enough free entries already)
689 uint32_t found = m_entries.size() - free_start;
690 uint32_t needed_entries = amount + 1;
691 uint32_t additional_entries = 0;
692 if (needed_entries > found)
693 additional_entries = needed_entries - found;
694
695 // Find the length of the current cluster chain
696 uint32_t total_entries = m_entries.size() + additional_entries;
697 uint32_t total_clusters = (total_entries * sizeof(dir_entry_t)) /
698 (m_volume->bpb.bytes_per_sector * m_volume->bpb.sectors_per_cluster);
699
700 // Expand the cluster chain if needed
701 if (total_clusters > m_current_cluster_length) {
702 m_volume->allocate_cluster(m_last_cluster, total_clusters - m_current_cluster_length);
703 m_current_cluster_length = total_clusters;
704 }
705
706 // Expand the directory to fit the remaining entries
707 for (int i = 0; i < additional_entries; ++i) {
708 dir_entry_t free = {};
709 free.name[0] = (uint8_t) DirectoryEntryType::FREE;
710 m_entries.push_back(free);
711 }
712
713 // Write the updated end of entries
714 m_entries[m_entries.size() - 1].name[0] = (uint8_t) DirectoryEntryType::LAST;
715 update_entry_on_disk(m_entries.size() - 1);
716
717 return free_start;
718}

727Vector<long_file_name_entry_t> Fat32Directory::to_long_filenames(string name) {
728
729 size_t lfn_count = (name.length() + 12) / 13;
730 Vector<long_file_name_entry_t> lfn_entries;
731
732 // Create the long file name entries (in reverse order)
733 for (int i = lfn_count - 1; i >= 0; i--) {
734 // Create the long file name entry
735 long_file_name_entry_t lfn_entry;
736 lfn_entry.order = i + 1;
737 lfn_entry.attributes = 0x0F;
738 lfn_entry.type = 0;
739 lfn_entry.checksum = 0;
740
741 // If it is the last entry, set the last bit
742 if (i == lfn_entries.size() - 1)
743 lfn_entry.order |= 0x40;
744
745 // Set the name
746 for (int j = 0; j < 13; j++) {
747
748 // Get the character info (0xFFFF if out of bounds)
749 size_t char_index = i * 13 + j;
750 char c = (char_index < name.length()) ? name[char_index] : 0xFFFF;
751
752 // Set the character in the entry
753 if (j < 5)
754 lfn_entry.name1[j] = c;
755 else if (j < 11)
756 lfn_entry.name2[j - 5] = c;
757 else

759 }

774string Fat32Directory::parse_long_filename(long_file_name_entry_t *entry, const string &current) {
775
776 // Extract the long name from each part (in reverse order)
777 string current_long_name = "";
778 for (int i = 0; i < 13; i++) {
779
780 // Get the character (in utf8 encoding)
781 char c;
782 if (i < 5)
783 c = entry->name1[i] & 0xFF;
784 else if (i < 11)
785 c = entry->name2[i - 5] & 0xFF;
786 else
787 c = entry->name3[i - 11] & 0xFF;
788
789 // Padding / invalid or end of string
790 if (c == (char) 0xFF || c == '\0')
791 break;
792
793 // Add to the start as the entries are stored in reverse
794 current_long_name += string(c);
795 }
796
797 // Entry parsed (prepend name)
798 return current_long_name + current;
799}

801void Fat32Directory::read_from_disk() {
802
803 for (auto &file: m_files)
804 delete file;
805 m_files.clear();
806
807 for (auto &directory: m_subdirectories)
808 delete directory;
809 m_subdirectories.clear();
810
811 // Load the entries from the disk into memory
812 read_all_entries();
813
814 // Parse the entries
815 string long_name = "";
816 for (auto &entry: m_entries) {
817
818 // Skip free entries and volume labels
819 if (entry.name[0] == (uint8_t) DirectoryEntryType::FREE
820 || entry.attributes == (uint8_t) DirectoryEntryAttributes::FREE
821 || entry.attributes == (uint8_t) DirectoryEntryAttributes::VOLUME_ID)
822 continue;
823
824 // Extract the long name
825 if (entry.attributes == (uint8_t) DirectoryEntryAttributes::LONG_NAME) {
826 long_name = parse_long_filename((long_file_name_entry_t *) &entry, long_name);
827 continue;
828 }
829
830 bool is_directory = entry.attributes == (uint8_t) DirectoryEntryAttributes::DIRECTORY;
831
832 // Get the name of the entry
833 string name = long_name;
834 if (long_name == "") {
835 name = string(entry.name, 8);
836
837 // Add the extension
838 if (!is_directory)
839 name = name.strip() + "." + string(entry.extension, 3);
840 }
841
842 long_name = "";
843
844 // Get the starting cluster
845 uint32_t start_cluster = (entry.first_cluster_high << 16) | entry.first_cluster_low;
846
847 // Store the file or directory
848 if (is_directory)
849 m_subdirectories.push_back(new Fat32Directory(m_volume, start_cluster, name.strip()));
850 else
851 m_files.push_back(new Fat32File(m_volume, this, &entry, name));
852
853 }
854}
