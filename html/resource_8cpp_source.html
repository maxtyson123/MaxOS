161void BaseResourceRegistry::close_resource(Resource* resource, size_t flags) {
162
163 // Resource isn't stored in this registry
164 if(m_resources.find(resource->name()) == m_resources.end())
165 return;
166
167 m_resource_uses[resource->name()]--;
168
169 // Don't close a resource that has more processes using it
170 if(m_resource_uses[resource->name()])
171 return;
172
173 // Can safely close the resource
174 resource->close(flags);
175 m_resources.erase(resource->name());
176 m_resource_uses.erase(resource->name());
177 delete resource;
178}
