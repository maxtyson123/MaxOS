40void VideoGraphicsArray::write_registers(uint8_t *registers) {
41
42 // Move to the next register
43 m_misc_port.write(*(registers++));
44
45 // Set the sequencer registers
46 for (uint8_t i = 0; i < 5; i++) {
47 m_sequence_index_port.write(i);
48 m_sequence_data_port.write(*(registers++));
49 }
50
51 // Clear protection bit to enable writing to CR0-7
52 m_crtc_index_port.write(0x03);
53 crtc_data_port.write(crtc_data_port.read() | 0x80);
54 m_crtc_index_port.write(0x11);
55 crtc_data_port.write(crtc_data_port.read() | ~0x80);
56
57 // Ensure protection bit is set
58 registers[0x03] = registers[0x03] | 0x80;
59 registers[0x11] = registers[0x11] & ~0x80;
60
61 // Write the CRTC registers
62 for (uint8_t i = 0; i < 25; i++) {
63 m_crtc_index_port.write(i);
64 crtc_data_port.write(*(registers++));
65 }
66
67 // Write the graphics controller registers
68 for (uint8_t i = 0; i < 9; i++) {
69 m_graphics_controller_index_port.write(i);
70 m_graphics_controller_data_port.write(*(registers++));
71 }
72
73 // Write the attribute controller registers
74 for (uint8_t i = 0; i < 21; i++) {
75 m_attribute_controller_reset_port.read();
76 m_attribute_controller_index_port.write(i);
77 m_attribute_controller_write_port.write(*(registers++));
78 }
79
80 // Re-Lock CRTC and unblank display
81 m_attribute_controller_reset_port.read();
82 m_attribute_controller_index_port.write(0x20);
83
84}

136uint8_t *VideoGraphicsArray::get_frame_buffer_segment() {
137
138 // Optimise so that don't have to read and write to the port every time
139 return (uint8_t *) 0xA0000;
140
141 //read data from index number 6
142 m_graphics_controller_index_port.write(0x06);
143 uint8_t segmentNumber =
144 m_graphics_controller_data_port.read() &
145 (3 << 2); //Shift by 2 as only interested in bits 3 & 4 (& 3 so all the other bits are removed)
146 switch (segmentNumber) {
147 default:
148 case 0 << 2:
149 return (uint8_t *) nullptr;
150 case 1 << 2:
151 return (uint8_t *) 0xA0000;
152 case 2 << 2:
153 return (uint8_t *) 0xB0000;
154 case 3 << 2:
155 return (uint8_t *) 0xB8000;
156 }
157}
