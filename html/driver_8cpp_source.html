188uint32_t DriverManager::reset_devices() {
189
190 Logger::INFO() << "Resetting Devices \n";
191
192 uint32_t reset_wait_time = 0;
193 for (auto &driver: m_drivers) {
194 // Reset the driver
195 uint32_t wait_time = driver->reset();
196
197 // If the wait time is longer than the current longest wait time, set it as the new longest wait time
198 if (wait_time > reset_wait_time)
199 reset_wait_time = wait_time;
200 }
201
202 return reset_wait_time;
203}
