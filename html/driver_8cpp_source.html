171uint32_t DriverManager::reset_devices() {
172
173 Logger::INFO() << "Resetting Devices \n";
174
175 uint32_t resetWaitTime = 0;
176 for (auto &driver: m_drivers) {
177 // Reset the driver
178 uint32_t waitTime = driver->reset();
179
180 // If the wait time is longer than the current longest wait time, set it as the new longest wait time
181 if (waitTime > resetWaitTime)
182 resetWaitTime = waitTime;
183 }
184
185 return resetWaitTime;
186}
