175uint32_t DriverManager::reset_devices() {
176
177 Logger::INFO() << "Resetting Devices \n";
178
179 uint32_t resetWaitTime = 0;
180 for (auto &driver: m_drivers) {
181 // Reset the driver
182 uint32_t waitTime = driver->reset();
183
184 // If the wait time is longer than the current longest wait time, set it as the new longest wait time
185 if (waitTime > resetWaitTime)
186 resetWaitTime = waitTime;
187 }
188
189 return resetWaitTime;
190}
