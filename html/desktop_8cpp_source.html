93void Desktop::internal_invalidate(common::Rectangle<int32_t>& area, Vector<Rectangle<int32_t>>::iterator start,
94 Vector<Rectangle<int32_t>>::iterator stop) {
95
96 // Loop through the invalid rectangles
97 for (Vector<Rectangle<int32_t>>::iterator invaild_rect = start; invaild_rect != stop; invaild_rect++) {
98
99 // Check if the area intersects with the invalid rectangle
100 if (!area.intersects(*invaild_rect))
101 continue;
102
103 // Get the parts of the area that are covered by the invalid rectangle
104 Vector<Rectangle<int32_t>> coveredAreas = area.subtract(*invaild_rect);
105
106 // Invalidate the covered areas
107 for (auto& coveredArea: coveredAreas)
108 internal_invalidate(coveredArea, invaild_rect + 1, stop);
109
110 // The entire area will be invalidated by now
111 return;
112
113 }
114
115 // Add the area to the invalid areas, store where it was added
116 Vector<Rectangle<int32_t>>::iterator vectorPosition = m_invalid_areas.push_back(area);
117
118 // If the m_position is the last item then the invalidation buffer is full
119 if (vectorPosition == m_invalid_areas.end()) {
120
121 // Invalidate the entire desktop
122 m_invalid_areas.clear();
123 Widget::invalidate();
124
125 }
126}

157
158 // If the mouse is over the widget then send a mouse leave event to the child widget as it is no longer under the mouse

166 if (underMouse)

175void Desktop::on_time(common::Time const&) {
176
177 // Check if anything is invalid and needs to be redrawn
178 if (m_invalid_areas.empty())
179 return;
180
181 // Erase the mouse cursor
182 invert_mouse_cursor();
183
184 // Loop through the invalid areas
185 while (!m_invalid_areas.empty()) {
186
187 // Redraw the m_first_memory_chunk area
188 Rectangle<int32_t> invalidArea = *(m_invalid_areas.begin());
189 m_invalid_areas.pop_front();
190 draw(m_graphics_context, invalidArea);
191
192 }
193
194 // Can now draw the mouse cursor
195 invert_mouse_cursor();
196
197}

218void Desktop::on_mouse_move_event(int8_t x, int8_t y) {
219
220 // Calculate the m_position of the mouse on the desktop
221 Rectangle<int32_t> desktopPosition = position();
222 int32_t newMouseX = m_mouse_x + x;
223 int32_t newMouseY = m_mouse_y + y;
224
225 // Restrain the mouse to the desktop
226 if (newMouseX < 0) newMouseX = 0;
227 if (newMouseY < 0) newMouseY = 0;
228 if (newMouseX > desktopPosition.width) newMouseX = desktopPosition.width - 1;
229 if (newMouseY > desktopPosition.height) newMouseY = desktopPosition.height - 1;
230
231 // Remove the old cursor from the screen as it will be redrawn in the new m_position
232 invert_mouse_cursor();
233
234 // If a widget is being dragged then pass the event to it
235 if (m_dragged_widget != nullptr)
236 m_dragged_widget->on_mouse_move_event(newMouseX - m_mouse_x, newMouseY - m_mouse_y);
237
238 // Handle the mouse moving event (pass it to the widget that the mouse is over)
239 CompositeWidget::on_mouse_move_widget(m_mouse_x, m_mouse_y, newMouseX,
240 newMouseY);
241
242 // Update the mouse m_position
243 m_mouse_x = newMouseX;
244 m_mouse_y = newMouseY;
245
246 // Draw the new cursor
247 invert_mouse_cursor();
248}
