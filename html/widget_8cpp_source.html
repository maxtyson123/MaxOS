136void Widget::resize(int32_t width, int32_t height) {
137
138 // Restrict the width and height to the minimum and maximum values
139 if (width < (int) m_min_width) width = m_min_width;
140 if (height < (int) m_min_height) height = m_min_height;
141 if (width > (int) m_max_width) width = m_max_width;
142 if (height > (int) m_max_height) height = m_max_height;
143
144 // Store the old position, set the new position
145 Rectangle<int32_t> old_position = m_position;
146 m_position.width = width;
147 m_position.height = height;
148
149 // Find the areas that need to be redrawn by subtracting the old position from the new position, and vice versa
150 Vector<Rectangle<int32_t>> invalid_areas_old = old_position.subtract(m_position);
151 Vector<Rectangle<int32_t>> invalid_areas_new = m_position.subtract(old_position);
152
153 // Right and Bottom require to be fully invalidated TODO: Fix this hack
154 if (m_position.width > old_position.width || m_position.height > old_position.height ||
155 old_position.width > m_position.width || old_position.height > m_position.height) {
156 invalidate();
157 return;
158 }
159
160 //Loop through the areas that need to be redrawn and invalidate them
161 for (auto& area: invalid_areas_old)
162 invalidate(area);
163
164 for (auto& area: invalid_areas_new)
165 invalidate(area);
166
167}

317void CompositeWidget::draw(GraphicsContext* gc, Rectangle<int32_t>& area, Vector<Widget*>::iterator start) {
318
319 // Draw the widget
320 Widget::draw(gc, area);
321
322 // Get the area of the widget
323 Rectangle<int32_t> own_area = position();
324
325 //Note: has to use iterator as the start is not necessarily the m_first_memory_chunk child
326 for (Vector<Widget*>::iterator child_widget = start; child_widget != m_children.end(); child_widget++) {
327
328 Rectangle<int32_t> child_area = (*child_widget)->position();
329
330 // Check if the child is in the area that needs to be redrawn
331 if (area.intersects(child_area)) {
332
333 // Get the area that needs to be redrawn
334 Rectangle<int32_t> rectangle = area.intersection(child_area);
335
336 // Translate the area so that it is relative to the child
337 rectangle.left -= child_area.left;
338 rectangle.top -= child_area.top;
339
340 // Draw the child
341 (*child_widget)->draw(gc, rectangle);
342
343 // Draw what is left of the area that needs to be redrawn
344 Vector<Rectangle<int32_t>> rest_draw_area = area.subtract(child_area);
345 for (auto& rest_area_part: rest_draw_area)
346 draw(gc, rest_area_part, child_widget + 1);
347
348 // Return as the entire area has now been drawn
349 return;
350 }
351 }
352
353 // Now draw the widget itself
354 draw_self(gc, area);
355}

385void CompositeWidget::on_mouse_enter_widget(uint32_t toX, uint32_t toY) {
386
387 for (auto& child_widget: m_children) {
388
389 // Check if the mouse is in the child
390 Rectangle<int32_t> child_area = child_widget->position();
391 if (child_area.contains(toX, toY)) {
392
393 // Get the position of the mouse relative to the child
394 uint32_t child_x = toX - child_area.left;
395 uint32_t child_y = toY - child_area.top;
396
397 // Call the child's on_mouse_enter_widget function
398 child_widget->on_mouse_enter_widget(child_x, child_y);
399
400 // Break as the event has been handled
401 break;
402 }
403 }

441void CompositeWidget::on_mouse_move_widget(uint32_t fromX, uint32_t fromY, uint32_t toX, uint32_t toY) {
442
443 Widget* left_child = nullptr;
444 Widget* entered_child = nullptr;
445
446 for (auto& child_widget: m_children) {
447
448 // Check if the mouse is in the child
449 Rectangle<int32_t> child_area = child_widget->position();
450 bool mouse_in_from = child_area.contains(fromX, fromY);
451 bool mouse_in_to = child_area.contains(toX, toY);
452
453 // If the mouse started in the child
454 if (mouse_in_from) {
455
456 // The mouse moved out of the child
457 if (!mouse_in_to) {
458 left_child = child_widget;
459 continue;
460 }
461
462 // Mouse still in the child
463 child_widget->on_mouse_move_widget(fromX, fromY, toX, toY);
464
465 } else {
466
467 // Mouse moved into the child
468 if (mouse_in_to)
469 entered_child = child_widget;
470 }
471
472 // Pass the events to the child
473 if (left_child != nullptr)
474 left_child->on_mouse_leave_widget(fromX, fromY);
475
476 if (entered_child != nullptr)
477 entered_child->on_mouse_enter_widget(toX, toY);
478 }
