157void Widget::resize(int32_t width, int32_t height) {
158
159 // Restrict the width and height to the minimum and maximum values
160 if (width < (int) m_min_width) width = m_min_width;
161 if (height < (int) m_min_height) height = m_min_height;
162 if (width > (int) m_max_width) width = m_max_width;
163 if (height > (int) m_max_height) height = m_max_height;
164
165 // Store the old position, set the new position
166 Rectangle<int32_t> old_position = m_position;
167 m_position.width = width;
168 m_position.height = height;
169
170 // Find the areas that need to be redrawn by subtracting the old position from the new position, and vice versa
171 Vector<Rectangle<int32_t>> invalid_areas_old = old_position.subtract(m_position);
172 Vector<Rectangle<int32_t>> invalid_areas_new = m_position.subtract(old_position);
173
174 // Right and Bottom require to be fully invalidated
175 if (m_position.width > old_position.width || m_position.height > old_position.height ||
176 old_position.width > m_position.width || old_position.height > m_position.height) {
177 invalidate();
178 return;
179 }
180
181 //Loop through the areas that need to be redrawn and invalidate them
182 for (auto& area: invalid_areas_old)
183 invalidate(area);
184
185 for (auto& area: invalid_areas_new)
186 invalidate(area);
187
188}

346void CompositeWidget::draw(GraphicsContext* gc, Rectangle<int32_t>& area, Vector<Widget*>::iterator start) {
347
348 // Draw the widget
349 Widget::draw(gc, area);
350
351 // Get the area of the widget
352 Rectangle<int32_t> own_area = position();
353
354 //Note: has to use iterator as the start is not necessarily the m_first_memory_chunk child
355 for (Vector<Widget*>::iterator child_widget = start; child_widget != m_children.end(); child_widget++) {
356
357 Rectangle<int32_t> child_area = (*child_widget)->position();
358
359 // Check if the child is in the area that needs to be redrawn
360 if (area.intersects(child_area)) {
361
362 // Get the area that needs to be redrawn
363 Rectangle<int32_t> rectangle = area.intersection(child_area);
364
365 // Translate the area so that it is relative to the child
366 rectangle.left -= child_area.left;
367 rectangle.top -= child_area.top;
368
369 // Draw the child
370 (*child_widget)->draw(gc, rectangle);
371
372 // Draw what is left of the area that needs to be redrawn
373 Vector<Rectangle<int32_t>> rest_draw_area = area.subtract(child_area);
374 for (auto& rest_area_part: rest_draw_area)
375 draw(gc, rest_area_part, child_widget + 1);
376
377 // Return as the entire area has now been drawn
378 return;
379 }
380 }
381
382 // Now draw the widget itself
383 draw_self(gc, area);
384}

470void CompositeWidget::on_mouse_move_widget(uint32_t fromX, uint32_t fromY, uint32_t toX, uint32_t toY) {
471
472 Widget* left_child = nullptr;
473 Widget* entered_child = nullptr;
474
475 for (auto& child_widget: m_children) {
476
477 // Check if the mouse is in the child
478 Rectangle<int32_t> child_area = child_widget->position();
479 bool mouse_in_from = child_area.contains(fromX, fromY);
480 bool mouse_in_to = child_area.contains(toX, toY);
481
482 // If the mouse started in the child
483 if (mouse_in_from) {
484
485 // The mouse moved out of the child
486 if (!mouse_in_to) {
487 left_child = child_widget;
488 continue;
489 }
490
491 // Mouse still in the child
492 child_widget->on_mouse_move_widget(fromX, fromY, toX, toY);
493
494 } else {
495
496 // Mouse moved into the child
497 if (mouse_in_to)
498 entered_child = child_widget;
499 }
500
501 // Pass the events to the child
502 if (left_child != nullptr)
503 left_child->on_mouse_leave_widget(fromX, fromY);
504
505 if (entered_child != nullptr)
506 entered_child->on_mouse_enter_widget(toX, toY);
507 }
