297Directory* FileSystem::get_directory(const string &path) {
298
299 // Check if the path is the root
300 if (path == "/")
301 return root_directory();
302
303 // Recursively open the directory
304 Directory* directory = root_directory();
305 string directory_path = path;
306 while (directory_path.length() > 0) {
307
308 // Get the name of the directory
309 string directory_name = Path::top_directory(directory_path);
310
311 // Open the directory
312 Directory* subdirectory = directory->open_subdirectory(directory_name);
313 if (!subdirectory)
314 return nullptr;
315
316 // Set the new directory
317 directory = subdirectory;
318
319 // Get the path to the next directory
320 directory_path = directory_path.substring(directory_name.length() + 1, directory_path.length() - directory_name.length() - 1);
321 }
322
323 return directory;
324}
