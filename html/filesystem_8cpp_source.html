301Directory* FileSystem::get_directory(const string &path) {
302
303 // Check if the path is the root
304 if (path == "/")
305 return root_directory();
306
307 // Recursively open the directory
308 Directory* directory = root_directory();
309 string directory_path = path;
310 while (directory_path.length() > 0) {
311
312 // Get the name of the directory
313 string directory_name = Path::top_directory(directory_path);
314
315 // Open the directory
316 Directory* subdirectory = directory->open_subdirectory(directory_name);
317 if (!subdirectory)
318 return nullptr;
319
320 // Set the new directory
321 directory = subdirectory;
322
323 // Get the path to the next directory
324 directory_path = directory_path.substring(directory_name.length() + 1, directory_path.length() - directory_name.length() - 1);
325 }
326
327 return directory;
328}
