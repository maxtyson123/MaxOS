108uint32_t PeripheralComponentInterconnectController::read(uint16_t bus, uint16_t device, uint16_t function,
109 uint32_t register_offset) {
110
111 // Calculate the id
112 uint32_t id = 0x1 << 31
113 | ((bus & 0xFF) << 16)
114 | ((device & 0x1F) << 11)
115 | ((function & 0x07) << 8)
116 | (register_offset & 0xFC);
117 m_command_port.write(id);
118
119 // Read the data from the port
120 uint32_t result = m_data_port.read();
121 return result >> (8 * (register_offset % 4));
122
123}

168void PeripheralComponentInterconnectController::select_drivers(DriverSelectorEventHandler* handler) {
169
170 for (int bus = 0; bus < 8; ++bus) {
171 for (int device = 0; device < 32; ++device) {
172
173 int numFunctions = (device_has_functions(bus, device)) ? 8 : 1;
174
175 for (int function = 0; function < numFunctions; ++function) {
176
177 // Get the device descriptor, if the vendor id is 0x0000 or 0xFFFF, the device is not present/ready
178 PeripheralComponentInterconnectDeviceDescriptor deviceDescriptor = get_device_descriptor(bus, device,
179 function);
180 if (deviceDescriptor.vendor_id == 0x0000 || deviceDescriptor.vendor_id == 0x0001 ||
181 deviceDescriptor.vendor_id == 0xFFFF)
182 continue;
183
184 // Get the earliest port number
185 for (int barNum = 5; barNum >= 0; barNum--) {
186 BaseAddressRegister bar = get_base_address_register(bus, device, function, barNum);
187 if (bar.address && (bar.type == BaseAddressRegisterType::InputOutput))
188 deviceDescriptor.port_base = (uint64_t) bar.address;
189 }
190
191 Logger::DEBUG() << "DEVICE FOUND: " << deviceDescriptor.get_type() << " - ";
192
193 // Select the driver and print information about the device
194 Driver* driver = get_driver(deviceDescriptor);
195 if (driver != nullptr) {
196 handler->on_driver_selected(driver);
197 Logger::Out() << driver->vendor_name() << " " << driver->device_name();
198 } else {
199 list_known_device(deviceDescriptor);
200 }
201
202 Logger::Out() << "\n";
203 }
204 }
205 }
206}

216PeripheralComponentInterconnectDeviceDescriptor PeripheralComponentInterconnectController::get_device_descriptor(uint16_t bus, uint16_t device, uint16_t function) {
217
218 PeripheralComponentInterconnectDeviceDescriptor result;
219
220 result.bus = bus;
221 result.device = device;
222 result.function = function;
223
224 result.vendor_id = read(bus, device, function, 0x00);
225 result.device_id = read(bus, device, function, 0x02);
226
227 result.class_id = read(bus, device, function, 0x0B);
228 result.subclass_id = read(bus, device, function, 0x0A);
229 result.interface_id = read(bus, device, function, 0x09);
230
231 result.revision = read(bus, device, function, 0x8);
232 result.interrupt = read(bus, device, function, 0x3C);
233
234 return result;
235}

428BaseAddressRegister PeripheralComponentInterconnectController::get_base_address_register(uint16_t bus, uint16_t device, uint16_t function, uint16_t bar) {
429
430 BaseAddressRegister result{};
431
432 // Only types 0x00 (normal devices) and 0x01 (PCI-to-PCI bridges) are supported:
433 uint32_t headerType = read(bus, device, function, 0x0E);
434 if (headerType & 0x3F)
435 return result;
436
437 // read the base address register
438 uint64_t bar_value = read(bus, device, function, 0x10 + 4 * bar);
439 result.type = (bar_value & 0x1) ? BaseAddressRegisterType::InputOutput : BaseAddressRegisterType::MemoryMapped;
440 result.address = (uint8_t*) (bar_value & ~0xF);
441
442 // Read the size of the base address register
443 write(bus, device, function, 0x10 + 4 * bar, 0xFFFFFFF0 | (int) result.type);
444 result.size = read(bus, device, function, 0x10 + 4 * bar);
445 result.size = (~result.size | 0xF) + 1;
446
447 // Restore the original value of the base address register
448 write(bus, device, function, 0x10 + 4 * bar, bar_value);
449
450 return result;
451}
