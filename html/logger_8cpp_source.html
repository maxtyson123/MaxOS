213void Logger::printf(char const *format, ...) {
214
215 // Create a pointer to the data
216 va_list parameters;
217 va_start(parameters, format);
218
219 // Loop through the format string
220 for (; *format != '\0'; format++) {
221
222 // If it is not a %, print the character
223 if (*format != '%') {
224 write_char(*format);
225 continue;
226 }
227
228 // Move to the next character
229 format++;
230 switch (*format) {
231 case 'd': {
232 // Print a decimal
233 int number = va_arg (parameters, int);
234 write_int(number);
235 break;
236 }
237 case 'x': {
238 // Print a hex
239 uint64_t number = va_arg (parameters, uint64_t);
240 write_hex(number);
241 break;
242 }
243 case 's': {
244 // Print a string
245 char *str = va_arg (parameters, char*);
246 write(str);
247 break;
248 }
249 }
250 }
