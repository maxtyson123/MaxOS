126 void InterruptManager::set_interrupt_descriptor_table_entry(uint8_t interrupt, void (*handler)(), uint8_t descriptor_privilege_level)
127 {
128 
129  // Get the address of the handler and the entry in the IDT
130  uint64_t handler_address = (uint64_t)handler;
131  InterruptDescriptor* interrupt_descriptor = &s_interrupt_descriptor_table[interrupt];
132 
133  // Set the handler address
134  interrupt_descriptor->address_low_bits = handler_address & 0xFFFF;
135  interrupt_descriptor->address_mid_bits = (handler_address >> 16) & 0xFFFF;
136  interrupt_descriptor->address_high_bits = (handler_address >> 32) & 0xFFFFFFFF;
137 
138  // Set the kernel code segment offset
139  interrupt_descriptor->segment_selector = 0x08;
140 
141  // Disable IST
142  interrupt_descriptor->ist = 0;
143 
144  // Set the flags (Trap Gate, Present and the Descriptor Privilege Level)
145  interrupt_descriptor->flags = 0b1110 | ((descriptor_privilege_level & 0b11) << 5) | (1 << 7);
146 }

248 void InterruptManager::page_fault(system::cpu_status_t *status) {
249  bool present = (status ->error_code & 0x1) != 0; // Bit 0: Page present flag
250  bool write = (status ->error_code & 0x2) != 0; // Bit 1: Write operation flag
251  bool user_mode = (status ->error_code & 0x4) != 0; // Bit 2: User mode flag
252  bool reserved_write = (status ->error_code & 0x8) != 0; // Bit 3: Reserved bit write flag
253  bool instruction_fetch = (status ->error_code & 0x10) != 0; // Bit 4: Instruction fetch flag (on some CPUs)
254  uint64_t faulting_address;
255  asm volatile("movq %%cr2, %0" : "=r" (faulting_address));
256 
257  ASSERT(false, "Page Fault (0x%x): present: %s, write: %s, user-mode: %s, reserved write: %s, instruction fetch: %s\n",
258  faulting_address, (present ? "Yes" : "No"), (write ? "Yes" : "No"), (user_mode ? "Yes" : "No"), (reserved_write ? "Yes" : "No"), (instruction_fetch ? "Yes" : "No"));
259 }
