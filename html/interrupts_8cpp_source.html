13InterruptHandler::InterruptHandler(uint8_t interrupt_number, int64_t redirect, uint64_t redirect_index)
14: m_interrupt_number(interrupt_number)
15{
16
17 // Get the interrupt manager
18 InterruptManager* interrupt_manager = InterruptManager::s_active_interrupt_manager;
19 ASSERT(interrupt_manager != nullptr, "No active interrupt manager");
20
21 // Register the handler
22 interrupt_manager->set_interrupt_handler(m_interrupt_number, this);
23
24 // Not all interrupts need redirecting
25 if (redirect == -1)
26 return;
27
28 // Redirect a legacy interrupt to an interrupt the kernel expects
29 IOAPIC* io_apic = interrupt_manager->active_apic()->io_apic();
30 interrupt_redirect_t temp = {
31 .type = (uint8_t) redirect,
32 .index = (uint8_t) redirect_index,
33 .interrupt = m_interrupt_number,
34 .destination = 0x00,
35 .flags = 0x00,
36 .mask = false,
37 };
38 io_apic->set_redirect(&temp);
39}

152void InterruptManager::set_interrupt_descriptor_table_entry(uint8_t interrupt, void (* handler)(), uint8_t descriptor_privilege_level) {
153
154 // Get the address of the handler and the entry in the IDT
155 auto handler_address = (uint64_t) handler;
156 InterruptDescriptor* interrupt_descriptor = &s_interrupt_descriptor_table[interrupt];
157
158 // Set the handler address
159 interrupt_descriptor->address_low_bits = handler_address & 0xFFFF;
160 interrupt_descriptor->address_mid_bits = (handler_address >> 16) & 0xFFFF;
161 interrupt_descriptor->address_high_bits = (handler_address >> 32) & 0xFFFFFFFF;
162
163 // Set the kernel code segment offset
164 interrupt_descriptor->segment_selector = 0x08;
165
166 // Disable IST
167 interrupt_descriptor->ist = 0;
168
169 // Set the flags (Trap Gate, Present and the Descriptor Privilege Level)
170 interrupt_descriptor->flags = 0b1110 | ((descriptor_privilege_level & 0b11) << 5) | (1 << 7);
171}

209system::cpu_status_t* InterruptManager::HandleInterrupt(system::cpu_status_t* status) {
210
211 // Default Fault Handlers
212 switch (status->interrupt_number) {
213
214 case 0x7:
215 Logger::ERROR() << "Device Not Available: FPU Not Enabled\n";
216 CPU::prepare_for_panic(status);
217 CPU::PANIC("See above message for more information", status);
218 break;
219
220 case 0x0D:
221 return general_protection_fault(status);
222
223 case 0x0E:
224 return page_fault(status);
225 }
226
227 // If there is an interrupt manager handle interrupt
228 if (s_active_interrupt_manager != nullptr)
229 return s_active_interrupt_manager->handle_interrupt_request(status);
230
231 // CPU Can continue
232 return status;
233}

272cpu_status_t* InterruptManager::handle_interrupt_request(cpu_status_t* status) {
273
274 // Where to go afterward
275 cpu_status_t* new_status = status;
276
277 // If there is an interrupt manager, handle the interrupt
278 if (m_interrupt_handlers[status->interrupt_number] != nullptr)
279 new_status = m_interrupt_handlers[status->interrupt_number]->handle_interrupt(status);
280 else
281 Logger::WARNING() << "Interrupt " << (int) status->interrupt_number << " not handled\n";
282
283 // Send the EOI to the APIC
284 if (s_hardware_interrupt_offset <= status->interrupt_number && status->interrupt_number < s_hardware_interrupt_offset + 16)
285 m_apic->local_apic()->send_eoi();
286
287 // Return the status
288 return new_status;
289}

307cpu_status_t* InterruptManager::page_fault(system::cpu_status_t* status) {
308
309 // Extract the information about the fault
310 bool present = (status->error_code & 0x1) != 0;
311 bool write = (status->error_code & 0x2) != 0;
312 bool user_mode = (status->error_code & 0x4) != 0;
313 bool reserved_write = (status->error_code & 0x8) != 0;
314 bool instruction_fetch = (status->error_code & 0x10) != 0;
315 uint64_t faulting_address;
316 asm volatile("movq %%cr2, %0" : "=r" (faulting_address));
317
318 // Try kill the process so the system doesn't die
319 cpu_status_t* can_avoid = CPU::prepare_for_panic(status);
320 if (can_avoid != nullptr)
321 return can_avoid;
322
323 // Cant avoid it so halt the kernel
324 Logger::ERROR() << "Page Fault: (0x" << faulting_address << "): present: " << (present ? "Yes" : "No")
325 << ", write: " << (write ? "Yes" : "No") << ", user-mode: " << (user_mode ? "Yes" : "No")
326 << ", reserved write: " << (reserved_write ? "Yes" : "No") << ", instruction fetch: "
327 << (instruction_fetch ? "Yes" : "No") << "\n";
328 CPU::PANIC("See above message for more information", status);
329
330 // Probably should never get here
331 return status;
332}

340cpu_status_t* InterruptManager::general_protection_fault(system::cpu_status_t* status) {
341
342 uint64_t error_code = status->error_code;
343
344 // Try to avoid the panic
345 cpu_status_t* can_avoid = CPU::prepare_for_panic(status);
346 if (can_avoid != nullptr)
347 return can_avoid;
348
349 // Have to panic
350 Logger::ERROR() << "General Protection Fault: (0x" << status->rip << "): "
351 << (error_code & 0x1 ? "Protection-Exception" : "Not a Protection Exception") << "\n";
352 CPU::PANIC("See above message for more information", status);
353
354 // Probably should never get here
355 return status;
356}
