98bool ELF64::is_valid() const {
99
100 // Validate the magic number
101 for (size_t i = 0; i < 4; i++)
102 if (header()->identification[i] != ELF_MAGIC[i])
103 return false;
104
105 // Check if the elf is 64 bit
106 if (header()->identification[(int) ELFIdentification::Class] != (int) ELFClass::Bits64)
107 return false;
108
109 // Check if the elf is little endian
110 if (header()->identification[(int) ELFIdentification::Data] != (int) ELFData::LittleEndian)
111 return false;
112
113 // Check if the elf is version 1
114 if (header()->identification[(int) ELFIdentification::Version] != (int) ELFVersion::Current)
115 return false;
116
117 // Check if the elf is for the MaxOS platform
118 // if(header() -> identification[OSABI] != MaxOSABI)
119 // return false;
120
121 // Check if the elf is executable
122 if (header()->type != (int) ELFType::Executable)
123 return false;
124
125 // Check if the elf is for the x86_64 platform
126 if (header()->machine != (int) ELFMachine::x86_64)
127 return false;
128
129 // LGTM
130 return true;

137void ELF64::load_program_headers() const {
138
139 for (size_t i = 0; i < header()->program_header_count; i++) {
140
141 // Get the header information
142 elf_64_program_header_t* program_header = get_program_header(i);
143
144 // Only load headers that actually need loading
145 if (program_header->type != (int) ELFProgramType::Load)
146 continue;
147
148 // Allocate space at the requested address
149 void* address = MemoryManager::s_current_memory_manager->vmm()->allocate(program_header->virtual_address, program_header->memory_size, PRESENT | WRITE);
150 ASSERT(address != nullptr, "Failed to allocate memory for program header\n");
151
152 // Copy the program into memory at that address
153 memcpy(address, (void*) (m_elf_header_address + program_header->offset), program_header->file_size);
154
155 // Zero the rest of the memory if needed
156 size_t zero_size = program_header->memory_size - program_header->file_size;
157 memset((void*) ((uintptr_t) address + program_header->file_size), 0, zero_size);
158
159 // Once the memory has been copied can now mark the pages as read only etc
160 uint64_t flags = to_vmm_flags(program_header->flags);
161 PhysicalMemoryManager::s_current_manager->change_page_flags(address, flags, MemoryManager::s_current_memory_manager->vmm()->pml4_root_address());
162 }
163}
