41Event<KeyboardEvents> *KeyboardEventHandler::on_event(Event<KeyboardEvents> *event) {
42
43 switch (event->type) {
44
45 case KeyboardEvents::KEYDOWN:
46 this->on_key_down(((KeyDownEvent *) event)->key_code, ((KeyDownEvent *) event)->keyboard_state);
47 break;
48
49 case KeyboardEvents::KEYUP:
50 this->on_key_up(((KeyUpEvent *) event)->key_code, ((KeyUpEvent *) event)->keyboard_state);
51 break;
52
53 default:
54 break;
55 }
56
57 return event;
58}

74void KeyboardDriver::activate() {
75
76 // Wait for user to stop pressing key (this is for the start-up key e.g. hold 'F12' for boot menu or hold 'del' for bios )
77 while (m_command_port.read() & 0x1)
78 m_data_port.read();
79
80 // Enable keyboard interrupts
81 m_command_port.write(0xAE);
82
83 // Get the current state of the keyboard
84 m_command_port.write(0x20);
85 uint8_t status = (m_data_port.read() | 1) & ~0x10;
86
87 // Reset the keyboard
88 m_command_port.write(0x60);
89 m_data_port.write(status);
90
91 // Activate the keyboard
92 m_data_port.write(0xF4);
93}

149void KeyboardInterpreterEN_US::on_stream_read(uint8_t scan_code) {
150
151
152 // Check if the key was released
153 bool released = (scan_code & 0x80) && (m_current_extended_code_1
154 || (scan_code != 0xe1)) && (m_next_is_extended_code_0
155 || (scan_code != 0xe0));
156
157 // Clear the released bit
158 if (released)
159 scan_code &= ~0x80;
160
161 // Start of an extended scan code
162 if (scan_code == 0xe0) {
163 m_next_is_extended_code_0 = true;
164 return;
165 }
166
167 // Handle extended scan codes
168 ScanCodeType type = ScanCodeType::REGULAR;
169 if (m_next_is_extended_code_0) {
170
171 type = ScanCodeType::EXTENDED;
172 m_next_is_extended_code_0 = false;
173
174 // Check if the scan_code represents a shift key and return (fake shift)
175 if ((KeyboardInterpreterEN_US::KeyCodeEN_US) scan_code == KeyCodeEN_US::leftShift ||
176 (KeyboardInterpreterEN_US::KeyCodeEN_US) scan_code == KeyCodeEN_US::rightShift)
177 return;
178 }
179
180 // If the scan_code is 0xe1, set the e1Code flag to 1 and return
181 if (scan_code == 0xe1) {
182 m_current_extended_code_1 = 1;
183 return;
184 }
185
186 // If e1Code is 1, set e1Code to 2, store the scan_code in e1CodeBuffer, and return
187 if (m_current_extended_code_1 == 1) {
188 m_current_extended_code_1 = 2;
189 m_extended_code_1_buffer = scan_code;
190 return;
191 }
192
193 // If e1Code is 2, set keyType to 2, reset e1Code, and update e1CodeBuffer
194 if (m_current_extended_code_1 == 2) {
195 type = ScanCodeType::EXTENDED;
196 m_current_extended_code_1 = 0;
197 m_extended_code_1_buffer |= (((uint16_t) scan_code) << 8);
198 }
199
200 // Scan code value manipulations

202 bool should_be_upper_case = is_shifting != this->m_keyboard_state.caps_lock;
